185void Setup::computeTreeHeightLimits() {
186 const auto albero = Eng::Base::findByName(sceneRoot_, "albero");
187 const auto cone0 = std::dynamic_pointer_cast<Eng::Mesh>(Eng::Base::findByName(sceneRoot_, "Cone"));
188 const auto cone1 = std::dynamic_pointer_cast<Eng::Mesh>(Eng::Base::findByName(sceneRoot_, "Cone_001"));
189 const auto cone2 = std::dynamic_pointer_cast<Eng::Mesh>(Eng::Base::findByName(sceneRoot_, "Cone_002"));
190
191 if (!albero || (!cone0 && !cone2)) {
192 minTreeY_ = maxTreeY_ = pivotY_;
193 return;
194 }
195
196 auto processMesh = [&](const std::shared_ptr<Eng::Mesh>& mesh) {
197 const auto& verts = mesh->getVertexes();
198 glm::mat4 globalTransform = albero->getMatrix() * mesh->getMatrix();
199 for (const auto& v : verts) {
200 glm::vec4 worldV = globalTransform * glm::vec4(v, 1.0f);
201 if (worldV.y < minTreeY_) minTreeY_ = worldV.y;
202 if (worldV.y > maxTreeY_) maxTreeY_ = worldV.y;
203 }
204 };
205
206 if (cone0) processMesh(cone0);
207 if (cone2) processMesh(cone2);
208
209 if (minTreeY_ == std::numeric_limits<float>::max()) {
210 minTreeY_ = maxTreeY_ = pivotY_;
211 }
212}
