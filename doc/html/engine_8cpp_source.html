371void Base::render(const std::shared_ptr<Camera>& camera, const std::shared_ptr<List>& renderList,
372 const std::list<std::string>& excludedList) const
373{
374 renderList->render(camera->getViewMatrix());
375
376
377 // Crea una matrice per ridurre l'altezza delle ombre (appiattimento sul piano, z' = 0.05 * z)
378 const glm::mat4 shadowModelScaleMatrix = glm::scale(glm::mat4(1.0f), glm::vec3(1.0f, 0.05f, 1.0f));
379
380 for (const auto& instance : renderList->getRenderList())
381 {
382 // @brief Tenta il cast a Mesh
383 std::shared_ptr<Mesh> mesh = std::dynamic_pointer_cast<Mesh>(instance.getNodePtr());
384
385
386 if (mesh)
387 for (const auto& name : excludedList)
388 {
389 if (mesh->getName() == name)
390 mesh->setShadow(false);
391 }
392
393
394 // @brief Se è un Mesh e può proiettare ombre
395 if (mesh && mesh->getShadow())
396 {
397 const std::shared_ptr<Material> originalMaterial = mesh->getMaterial();
398
399 mesh->setMaterial(shadowMaterial_);
400
401 const glm::mat4 shadow_matrix = shadowModelScaleMatrix * instance.getNodeWorldMatrix();
402
403 glm::mat4 shadowModelViewMatrix = camera->getViewMatrix() * shadow_matrix;
404
405
407 mesh->render(shadowModelViewMatrix);
408
410 mesh->setMaterial(originalMaterial);
411 }
412 }
413
414 glClear(GL_DEPTH_BUFFER_BIT);
415
416
417
418}

622std::shared_ptr<Texture> Base::loadTextureFromFile(const std::string& path) noexcept
623{
624 FREE_IMAGE_FORMAT format = FreeImage_GetFileType(path.c_str(), 0);
625 if (format == FIF_UNKNOWN)
626 format = FreeImage_GetFIFFromFilename(path.c_str());
627
628 if (format == FIF_UNKNOWN)
629 {
630 std::cout << "ERROR: Unknown image format " << path << '\n';
631 return nullptr;
632 }
633
634 FIBITMAP* bitmap = FreeImage_Load(format, path.c_str());
635 if (!bitmap)
636 {
637 std::cout << "ERROR: Failed to load image " << path << '\n';
638 return nullptr;
639 }
640
641 FreeImage_FlipVertical(bitmap);
642
643 unsigned int width = FreeImage_GetWidth(bitmap);
644 unsigned int height = FreeImage_GetHeight(bitmap);
645 BYTE* data = FreeImage_GetBits(bitmap);
646
648 const unsigned int bpp = FreeImage_GetBPP(bitmap);
649 unsigned int channels = bpp / 8;
650
651 auto texture = std::make_shared<Texture>(data, width, height, channels);
652
653 const size_t p = path.find_last_of(pathSeparator);
654 texture->setName(path.substr(p + 1));
655
656 FreeImage_Unload(bitmap);
657 return texture;
658}

665void Base::bindTexturesToMaterials(const std::shared_ptr<Node>& root,
666 const std::vector<std::shared_ptr<Texture>>& textures) const noexcept
667{
668 if (!root || textures.empty())
669 {
670 return;
671 }
672
673 for (const auto& childNode : root->getChildren())
674 {
675 auto mesh = std::dynamic_pointer_cast<Mesh>(childNode);
676
677 if (mesh)
678 {
679 auto material = mesh->getMaterial();
680
681 if (material && !material->getTextureName().empty())
682 {
683 for (const auto& texture : textures)
684 {
685 if (texture->getName() == material->getTextureName())
686 {
687 material->setTexture(texture);
688 }
689 }
690 }
691 }
692
693 // Ricorsione
694 this->bindTexturesToMaterials(childNode, textures);
695 }

701void Base::loadAndBindTextures(const std::filesystem::path& projectDir_, const std::shared_ptr<Node>& root) const
702{
703 std::vector<std::shared_ptr<Texture>> textures;
704 const std::filesystem::path textureDir = projectDir_;
705
706 if (!std::filesystem::exists(textureDir) || !std::filesystem::is_directory(textureDir))
707 {
708 std::cerr << "ATTENZIONE: Cartella texture non valida." << std::endl;
709 return;
710 }
711
712 std::cout << "--- CARICAMENTO TEXTURE ---" << std::endl;
713 for (const auto& entry : std::filesystem::directory_iterator(textureDir))
714 {
715 if (entry.is_regular_file())
716 {
717 std::string ext = entry.path().extension().string();
718 std::transform(ext.begin(), ext.end(), ext.begin(), [](unsigned char c) { return std::tolower(c); });
719
720 if (ext == ".dds" || ext == ".png" || ext == ".jpg" || ext == ".jpeg" || ext == ".tga" || ext == ".bmp")
721 {
722 std::cout << "Loading: " << entry.path().filename() << "... ";
723 auto tex = loadTextureFromFile(entry.path().string());
724 if (tex)
725 {
726 textures.push_back(tex);
727 std::cout << "OK" << std::endl;
728 }
729 else
730 {
731 std::cout << "FALLITO" << std::endl;
732 }
733 }
734 }
735 }
736 std::cout << "Totale: " << textures.size() << std::endl;
737 bindTexturesToMaterials(root, textures);
738}
