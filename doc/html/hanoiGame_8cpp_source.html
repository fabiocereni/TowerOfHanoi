198 if (!pole) return nullptr;

200 auto disks = pole->getChildren();

217bool HanoiGame::isValidMove(std::shared_ptr<Eng::Node> destPole, std::shared_ptr<Eng::Node> diskToMove) {
218 auto topDiskDest = getTopDisk(destPole);
219
220 // Se il palo � vuoto, � sempre valido
221 if (!topDiskDest) return true;
222
223 // LOGICA CORRETTA:
224 // Poiche' "disco7" (piccolo) > "disco1" (grande) alfabeticamente:
225 // Il disco che muovo deve avere un nome "MAGGIORE" di quello che sta sotto.
226 // Esempio valido: Muovo "disco7" su "disco1" -> "disco7" > "disco1" (VERO)
227 // Esempio errato: Muovo "disco1" su "disco7" -> "disco1" > "disco7" (FALSO)
228
229 return diskToMove->getName() > topDiskDest->getName(); // <--- CAMBIA QUI DA < A >
230}

339float HanoiGame::getMeshHeight(const std::shared_ptr<Eng::Node>& node) {
340 auto mesh = std::dynamic_pointer_cast<Eng::Mesh>(node);
341 if (!mesh) return 0.0f; // Se non è una mesh (es. luce), altezza 0
342
343 float minY = std::numeric_limits<float>::max();
344 float maxY = std::numeric_limits<float>::lowest();
345
346 // Scansiona tutti i vertici per trovare l'estensione su Y
347 for (const auto& v : mesh->getVertexes()) {
348 if (v.y < minY) minY = v.y;
349 if (v.y > maxY) maxY = v.y;
350 }
351
352 // Calcoliamo lo spessore base
353 float height = maxY - minY;
354
355 // Moltiplichiamo per la scala Y della matrice locale (se l'oggetto è scalato)
356 // La lunghezza della seconda colonna della matrice (index 1) è la scala Y
357 float scaleY = glm::length(glm::vec3(node->getMatrix()[1]));
358
359 return height * scaleY;
360}
